#include<iostream>

#include<cstring>
#include<vector>

#include<stdio.h>

	ios::sync_with_stdio(false);

	int n;
	cin >> n;
	
	vector<int> anser(n);
	for (int i = 0; i < n; ++i) {
		scanf("%d", anser[i]);
	}

	sort(anser.begin(), anser.end());

	for (int i = 0; i < n; ++i) {
		cout << anser[i] << '\n';
	}
}
